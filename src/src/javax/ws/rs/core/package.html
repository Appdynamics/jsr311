Low-level interfaces and annotations used to create RESTful service resources. E.g.:

@URITemplate("widgets/{widgetid}")
public class WidgetResource extends WebResource {

  public void handleRequest(HTTPRequestContext requestContext, HTTPResponseContext responseContext) {
    if (requestContext.getHttpMethod().equals("GET")) {
      String replyStr = "<widget id='"+
        requestContext.getURIParameters().getFirst("widgetId")+"'/>";
      StringRepresentation representation = new StringRepresentation(replyStr,
        "application/widgets+xml");
      response.setRepresentation(new HttpResponse(representatin));
    }
    else ...
  }

}
